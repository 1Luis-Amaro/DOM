--Tabela para consultar atributos na tabela de interface
SELECT MSI.SEGMENT1,
       MSI.GLOBAL_ATTRIBUTE4,
       MSI.GLOBAL_ATTRIBUTE5,
       MSI.GLOBAL_ATTRIBUTE6,
       RILA.*
  FROM APPS.RA_INTERFACE_LINES_ALL RILA, APPS.MTL_SYSTEM_ITEMS MSI
 WHERE RILA.INVENTORY_ITEM_ID = MSI.INVENTORY_ITEM_ID
   AND RILA.WAREHOUSE_ID = MSI.ORGANIZATION_ID
   AND INTERFACE_LINE_ATTRIBUTE3 IN ('88454527');
