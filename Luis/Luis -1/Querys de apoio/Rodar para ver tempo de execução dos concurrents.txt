--Rodar para ver tempo de execução dos concurrents 


SELECT user_concurrent_program_name
      ,a.request_id
      ,phase_code
      ,status_code
      ,actual_start_date
      ,a.actual_completion_date
      ,argument_text
      ,a.*
FROM   apps.fnd_conc_req_summary_v a
WHERE  trunc(requested_start_date) BETWEEN trunc(SYSDATE - 1) AND
       trunc(SYSDATE - 1);
