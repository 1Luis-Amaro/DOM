SELECT RULE_ID,RULE_CODE,TAX_TYPE,TRIBUTO,BEGIN_EFFECTIVE_DATE,END_EFFECTIVE_DATE,
       TRIBUTATION_CODE,TRIBUTACAO,UTILIZATION_ID,UTILIZACAO,INACTIVE_DATE,
       SOURCE_STATE_ID,SOURCE_STATE,DESTINATION_STATE_ID,DESTINATION_STATE,
       DIFFER_TAX_FLAG,REDUCTION_PERCENT,INCREASE_PERCENT,EXEMPT_TAX_FLAG,
       SUSPENDED_TAX_FLAG,DIFFERENT_TAX,DIFFERENT_RECOVER_TAX_RATE,DIFFERENT_TAX_ICMS,
       INTERNAL_ICMS_RATE_DESTINATION,PISCOFINS_PRESUMED_CREDIT_FLAG,DEFERRED_TAX_PERC,
       CALC_BASIS_DEFERRED_FLAG,DESCRIPTION,ATTRIBUTE_CATEGORY 
  FROM cll_f189_inq_reg_acts_v 
 WHERE rule_id IN ( SELECT rule_id FROM cll_f189_fiscal_exceptions 
                     WHERE classification_id = NVL ('',classification_id) ) 
   and tax_type = nvl('',tax_type) 
   AND ( rule_id IN ( SELECT rule_id FROM cll_f189_gov_rule_org_rels 
                       WHERE organization_id = '' ) OR '' IS NULL )  
order by rule_code
