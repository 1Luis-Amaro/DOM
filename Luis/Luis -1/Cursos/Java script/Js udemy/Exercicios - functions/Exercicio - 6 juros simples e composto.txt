const jurosSimples = function (capitalInicial, taxaDeJuros, tempoDeAplicacao){

	return (capitalInicial + (capitalInicial * taxaDeJuros * tempoDeAplicacao))

}

console.log(jurosSimples(1000,0.10,60))
